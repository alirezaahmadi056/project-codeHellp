        private fun convertStreamToString(stream: InputStream): String {

            val bufferReader = BufferedReader(InputStreamReader(stream))
            var line: String
            val sb = StringBuilder()
            try {

                do {
                    line = bufferReader.readLine()
                    sb.append(line)
                } while (true)

            } catch (ex: Exception) {

            }

            return sb.toString()

        }
