fun IsInstallpackManger(context: Context, appName: String): Boolean {
        val pm = context.packageManager
        try {
            pm.getPackageInfo(appName, PackageManager.GET_ACTIVITIES)
            return true
        } catch (e: PackageManager.NameNotFoundException) {
            return false
        }

    }
